Query the latest price snapshot, best bid/ask price, and trading volume in the last 24 hours.
> **Cover: Spot / Linear contract / Inverse contract / Option**

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

### HTTP Request
GET `/v5/market/tickers`

### Request Parameters
| Parameter | Required | Type | Comments|
|:----- |:-------|:-----|------ |
|[category](../enum#category) |**true** |string |Product type. `spot`,`linear`,`inverse`,`option` |
|symbol |false |string |Symbol name |
|baseCoin |false |string |Base coin. `option` valid ONLY, either pass `baseCoin` or `symbol`|
|expDate |false |string |Expiry date. e.g., 25DEC22. for `option` ONLY|

### Response Parameters
| Parameter | Type | Comments|
|:----- |:-----|----- |
|category |string |Product type |
|list |array |Object |
|> symbol |string |Symbol name |
|> lastPrice |string |Last price |
|> markPrice |string |Mark price. _`spot` N/A_ |
|> indexPrice |string |Index price. _`spot` N/A_ |
|> underlyingPrice |string |Underlying price. _`option` ONLY_ |
|> bid1Price |string |Bid 1 price. _`option` N/A_ |
|> ask1Price |string |Ask 1 price. _`option` N/A_ |
|> bid1Size |string |Bid 1 size. _`option` N/A_ |
|> ask1Size |string |Ask 1 size. _`option` N/A_ |
|> openInterest |string |Open interest size. _`spot` N/A_ |
|> openInterestValue |string |Open interest value. _`linear`,`inverse` ONLY_ |
|> [lastTickDirection](../enum#lasttickdirection) |string |Direction of price change. _`linear`,`inverse` ONLY_ |
|> askIv |string |Ask 1 iv. _`option` ONLY_ |
|> askPrice |string |Ask 1 price. _`option` ONLY_ |
|> askSize |string |Ask 1 size. _`option` ONLY_ |
|> bidIv |string |Bid 1 iv. _`option` ONLY_ |
|> bidPrice |string |Bid 1 price. _`option` ONLY_ |
|> bidSize |string |Bid 1 size. _`option` ONLY_ |
|> markIv |string |Mark price iv. _`option` ONLY_ |
|> prevPrice1h |string |Market price an hour ago. _`linear`,`inverse` ONLY_ |
|> lowPrice24h |string |The lowest price in the last 24 hours |
|> highPrice24h |string |The highest price in the last 24 hours |
|> prevPrice24h |string |Market price 24 hours ago. _`option` N/A_ |
|> price24hPcnt |string |Percentage change of market price relative to 24h. _`option` N/A_ |
|> volume24h |string |Volume for 24h |
|> turnover24h |string |Turnover for 24h |
|> change24h |string |Change for 24h. _`option` ONLY_ |
|> totalTurnover |string |Total turnover. _`option` ONLY_ |
|> totalVolume |string |Total volume. _`option` ONLY_ |
|> basisRate |string |Basis rate. _`linear`,`inverse` ONLY_ |
|> nextFundingTime |string |Next funding timestamp (ms). _`linear`,`inverse` ONLY_ |
|> fundingRate |string |Funding rate. _`linear`,`inverse` ONLY_ |
|> deliveryTime |string |Delivery timestamp (ms). _`linear`,`inverse` ONLY_ |
|> deliveryFeeRate |string |Delivery fee rate. _`linear`,`inverse` ONLY_ |
|> predictedDeliveryPrice |string |Predicated delivery price. _`spot` N/A_ |
|> delta |string |Delta value. _`option` ONLY_ |
|> gamma |string |Gamma value. _`option` ONLY_ |
|> theta |string |Theta value. _`option` ONLY_ |
|> vega |string |Vega value. _`option` ONLY_ |
