Query a list of instruments of online trading pair.
> **Cover: Spot / USDT Perpetual / USDC Perpetual / Inverse Perpetual / Inverse Future / Option**

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

### HTTP Request
GET `/v5/market/instruments-info`

### Request Parameters
| Parameter | Required | Type | Comments|
|:----- |:-------|:-----|------ |
|[category](../enum#category) |<b>true</b> |string |Product type. `spot`,`linear`,`inverse`,`option`|
|symbol |false |string |Symbol name|
|baseCoin |false |string |Base coin. Valid for `linear`,`inverse`,`option`.|
|limit |false |integer |Limit for data size per page. [1,1000]. default: `500`|
|cursor |false |string |Cursor. Used to pagination|

### Response Parameters
| Parameter | Type | Comments|
|:----- |:-----|----- |
|category |string |Product type |
|list |array |Object |
|> symbol |string |Symbol name |
|> [contractType](../enum#contracttype) |string |Contract type. Valid for `linear`,`inverse`|
|> [status](../enum#status) |string |Symbol's trading status |
|> baseCoin |string |Base coin e.g., BTC in BTCUSDT |
|> quoteCoin |string |Quote coin e.g., USDT in BTCUSDT |
|> settleCoin |string |Settle coin e.g., USDC in BTCPERP |
|> optionsType |string |Option type. `Call`,`Put` |
|> launchTime |string |Symbol launch timestamp (ms) |
|> deliveryTime |string |Delivery timestamp (ms) |
|> deliveryFeeRate |string |Delivery fee rate |
|> fundingInterval |string |Funding rate interval (minute) |
|> leverageFilter |string |Object |
|>> minLeverage |string |Minimum leverage |
|>> maxLeverage |string |Maximum leverage |
|>> leverageStep |string |The minimal step to increase/reduce leverage |
|> priceFilter |string |Object |
|>> minPrice |string |Minimum order price |
|>> maxPrice |string |Maximum order price |
|>> tickSize |string |The minimal step to increase/reduce order price |
|> lotSizeFilter |string |Object |
|>> minOrderQty |string |Minimum order qty |
|>> maxOrderQty |string |Highest order qty |
|>> postOnlyMaxOrderQty |string |Maximum order qty for a single `PostOnly` order |
|>> qtyStep |string |The minimal step to increase/reduce order qty |
|> unifiedMarginTrade |boolean |Whether supports to trade under UMA/UTA |
|nextPageCursor |string |Cursor. Used to pagination |
